    \[\rho = \frac{actual~total~path~length}{ideal~total~path~length}\]

    \begin{align*}
      \rho = 1 & \implies \text{balanced}\\
      \rho < 1 & \implies \text{landscape}\\
      \rho > 1 & \implies \text{portrait}
    \end{align*}
  \end{minipage}
