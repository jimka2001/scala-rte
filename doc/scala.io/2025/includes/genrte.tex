def genRte(depth: Int): Rte = {
  val leaves = Seq(Sigma, EmptySeq, EmptySet)
  def rte = genRte(depth-1)
  val generators: = Seq(
      () => randomChoice(leaves),
      () => Singleton(genType()),
      () => Or(rte(), rte()),
      () => And(rte(), rte()),
      () => Cat(rte()),
      () => Star(rte()),
      () => Not(rte()))

  if (depth <= 0)
    Singleton(genType())
  else
    randomChoice(generators)()
}
