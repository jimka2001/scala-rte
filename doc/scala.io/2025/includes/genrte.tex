def genRte(depth: Int): Rte = {
  val leaves = Vector(Sigma, EmptySeq, EmptySet)
  def rte = genRte(depth-1)
  val generators: Seq[() => Rte] = Seq(
      () => leaves(random.nextInt(leaves.length)),
      () => Singleton(genType()),
      () => Or(rte(), rte()),
      () => And(rte(), rte()),
      () => Cat(rte()),
      () => Star(rte()),
      () => Not(rte()))

  if (depth <= 0)
    Singleton(genType())
  else
    generators(random.nextInt(generators.length))()
}
