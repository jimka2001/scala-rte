\begin{frame}{Example RTE Expression Tree}
  \usebox\rteast

  \medskip

  \centering

  \scalebox{0.7}{\input{rte-ast}}
  \begin{itemize}
  \item An RTE is an expression tree (abstract syntax
    tree, AST),
  \item Each node is an instance of class \code{Rte}.
  \item Leaf nodes interface to Scala classes via \code{Singleton(SAtomic(...))}.
  \end{itemize}
\end{frame}

\begin{frame}{\code{Rte} class quasi-AST}
  \scalebox{0.6}{\input{rte-adt}}

  \medskip

  \begin{itemize}
  \item RTE nodes (and leafs) are defined by the class hierarchy \code{Rte}
  \item The subclass \code{Singleton} interfaces to the type system via an instance of \code{SimpleTypeD}.
  \item \code{SimpleTypeD} is an abstraction of Scala/Java class (later)
  \item Not an ADT for a \Emph{silly} reason.
  \item Scala does not support cross-file \code{sealed} classes.
  \end{itemize}
\end{frame}
